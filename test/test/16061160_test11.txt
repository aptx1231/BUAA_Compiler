int Fibonacci(int n) {
    if (n == 0)
        return (0);
    if (n == 1)
        return (1);
    return (Fibonacci(n - 1) + Fibonacci(n - 2));
}

int fac(int n) {

        return (1);
       return (n * fac(n - 1));
}
